## polyfill.coreJsFeatures
:::tip

:::
+ 类型： `full` | `actual` | `stable` | `es`
+ 默认值： `stable`

## polyfill.browserslist
+ 浏览器 兼容版本
+ 类型：string[]
+ 默认值： ['chrome >= 87', 'edge >= 88', 'firefox >= 78', 'safari >= 14']

import {defineConfig} from '@empjs/cli'
export default defineConfig(store => {
  return {
    build:{

    }
